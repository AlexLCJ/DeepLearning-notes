##### 第一题：

```
从键盘任意输入一个整数m，若m不是素数，则对m进行质因数分解，并将m表示为质因数从小到大顺序排列的乘积形式输出，否则输出"It is a prime number"。例如，用户输入90时，程序输出90 = 2 * 3 * 3 * 5；用户输入17时，程序输出"It is a prime number"。
输入提示信息："Input m:"
输入格式："%d"
输出格式：
是素数时输出"It is a prime number\n"
否则输出用"%d = "，"%d * "
运行示例1：
Input m:90↙
90 = 2 * 3 * 3 * 5
运行示例2：
Input m:13↙
It is a prime number
```

```c
##### 

#include<stdio.h>
#include<math.h>
int sushu(int x);
void fenjie(int x);
int main(){
	int x;
	printf("Input m:");
	scanf("%d",&x);
	if(sushu(x)==1){
		printf("It is a prime number\n");
	}
	else{
		printf("%d = ",x);
		fenjie(x);
	}
	
	return 0;

}
int sushu(int x){
	int i;
	for(i=2;i<sqrt(x);i++){
		if(x%i==0){
			return -1;//不是素数 
		}
		

	}
	return 1;//是素数 

}
void fenjie(int x){
	int i;
	for(i=2;i<=sqrt(x);i++){
		if(x%i==0){
			printf("%d * ",i);
			fenjie(x/i);
			return ;
		}
	}
	printf("%d",x);
}
```

##### 第三题：

```
用函数编程统计平均成绩。要求在主函数中输入学生成绩，输入负数时,结束输入，调用子函数Average()计算平均成绩，并输出结果。
函数原型：int Average(int score[],int n)
****要求输入提示信息为："Input score:"
****输出格式要求为：
"Total students are %d\n"
"Average score is %d\n"
程序运行示例：
Input score:78
Input score:98
Input score:56
Input score:79
Input score:67
Input score:-1
Total students are 5
Average score is 75
```

```c
#include <stdio.h>
// 计算平均分数的函数
int Average(int score[], int n) {
	int total = 0;
	for (int i = 0; i < n; ++i) {
		total += score[i];
	}
	return total / n;
}

int main() {
	int scores[100];  // 假设最多有100个学生
	int input, count = 0;


// 输入学生成绩，直到输入负数为止
do {
    printf("Input score:");
	scanf_s("%d", &input);
	if (input >= 0) {
		scores[count] = input;
		count++;
	}
} while (input >= 0 && count < 100);

// 计算学生总数和平均分数，并输出结果
int average = Average(scores, count);
printf("Total students are %d\n", count);
printf("Average score is %d\n", average);

return 0;

}
```

##### 冒泡排序：（数组）

数组传参直接写数组名字：bubble_sort(arr);

有n个元素，就要进行n-1趟排序；

###### 	数组形式传参：(升序排列举例)

```c
###### #include<stdio.h>
void bublle_sort(int arr[],int sz);
int main(){
	int arr[]={9,8,7,6,5,4,3,2,1,0};
	int sz=sizeof(arr)/sizeof(arr[0]);//用于计算数组中有多少元素 
	bublle_sort(arr,sz);
	int i;
	for(i=0;i<sz;i++){
		printf("%d ",arr[i]);
	}
	return 0;
}

void bublle_sort(int arr[],int sz){
	int i;
	for(i=0;i<sz-1;i++){ 
		//一次排序，一共排序n-1次
		int j;//定义数组下标
		for(j=0;j<sz-1-i;j++){
			//j<sz-1-i 每次一次排序，就确定了最后一位数字的位置 第二次排序，就确定了最后两个位置的数字 
			if(arr[j]>arr[j+1]){
				//我需要得到升序排列，于是进行交换
				int t=arr[j];
				arr[j]=arr[j+1];
				arr[j+1]=t; 
			} 
		} 
	}
}

//常见错误：将int sz=sizeof(arr)/sizeof(arr[0]);放在void函数里面，调用数组的时候，表示一个指针，即指向首位，于是sz此时算出来只有1
```
